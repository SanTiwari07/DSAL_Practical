#include <iostream>
#include <string>
using namespace std;

struct Snode {
    string name;
    string phoneNumber;
    int roll;
    Snode *next;
    Snode *prev;
};

class DLinkedList {
public:
    Snode *head, *tail;

    DLinkedList() {
        head = nullptr;
        tail = nullptr;
    }

    // creates a node
    Snode *getNode(string name, string phoneNumber, int roll) {
        Snode *newNode = new Snode;
        newNode->name = name;
        newNode->phoneNumber = phoneNumber;
        newNode->roll = roll;
        newNode->next = NULL;
        newNode->prev = NULL;
        return newNode;
    }

    // insert menu
    void insertNode() {
        int roll, pos;
        string name, phoneNumber;
        cout << "\n\nEnter a position to insert data : ";
        cout << "\n1. At beginning";
        cout << "\n2. At end";
        cout << "\n3. At specific position : "<<endl;
        cin >> pos;
        cout << "\n Enter the name : ";
        cin >> name;
        cout << "\n Enter the phone number : ";
        cin >> phoneNumber;
        cout << "\n Enter the roll : ";
        cin >> roll;

        if (pos == 1)
            insertAtBeginning(name, phoneNumber, roll);
        else if (pos == 2)
            insertAtEnd(name, phoneNumber, roll);
        else
            insertAtPos(name, phoneNumber, roll, pos);
    }

    void insertAtBeginning(string name, string phoneNumber, int roll) {
        Snode *temp = getNode(name, phoneNumber, roll);
        if (head == NULL) {
            head = tail = temp;
        } else {
            temp->next = head;
            head->prev = temp;
            head = temp;
        }
    }

    void insertAtEnd(string name, string phoneNumber, int roll) {
        Snode *temp = getNode(name, phoneNumber, roll);
        if (tail == NULL) {
            head = tail = temp;
        } else {
            temp->prev = tail;
            tail->next = temp;
            tail = temp;
        }
    }

    int getCount() {
        int count = 0;
        Snode *temp = head;
        while (temp != NULL) {
            count++;
            temp = temp->next;
        }
        return count;
    }

    void insertAtPos(string name, string phoneNumber, int roll, int pos) {
        int count = getCount();
        if (pos < 1 || pos > count + 1) {
            cout << "Enter a valid position (1 to " << count + 1 << ")\n";
            return;
        }
        if (pos == 1) {
            insertAtBeginning(name, phoneNumber, roll);
        } else if (pos == count + 1) {
            insertAtEnd(name, phoneNumber, roll);
        } else {
            Snode *newNode = getNode(name, phoneNumber, roll);
            Snode *temp = head;
            for (int i = 1; i < pos - 1; i++) {
                temp = temp->next;
            }
            newNode->next = temp->next;
            newNode->prev = temp;
            temp->next->prev = newNode;
            temp->next = newNode;
        }
    }

    void display() {
        Snode *temp = head;
        if (head == NULL) {
            cout << "List is empty!!";
        } else {
            cout << "\n\nUser Data : " << endl;
            cout << "Name\tPhoneNumber\tRoll-No";
            while (temp != NULL) {
                cout << "\n" << temp->name << "\t" << temp->phoneNumber << "\t" << temp->roll;
                temp = temp->next;
            }
        }
        cout << endl;
    }

    void searchNode() {
        string field;
        int roll;
        cout << "\nEnter a field to search : ";
        cin >> field;
        cout << "\nEnter roll (else -1): ";
        cin >> roll;

        Snode *temp = head;
        while (temp != NULL) {
            if (temp->name == field || temp->phoneNumber == field || (roll != -1 && temp->roll == roll)) {
                cout << "\nContact found!!\n";
                cout << "Name : " << temp->name << "\nPhone Number : " << temp->phoneNumber << "\nRoll-No : " << temp->roll << endl;
                return;
            }
            temp = temp->next;
        }
        cout << "\nContact not found!!";
    }

    void deleteNode() {
        int pos;
        cout << "\nEnter position to delete (Enter -1 for last): ";
        cin >> pos;
        int count = getCount();

        if (pos == 1) {
            deleteAtFirst();
        } else if (pos == count || pos == -1) {
            deleteAtLast();
        } else {
            deleteAtPos(pos);
        }
    }

    void deleteAtFirst() {
        if (head == NULL) {
            cout << "List is empty!!\n";
            return;
        }
        Snode *temp = head;
        head = head->next;
        if (head != NULL)
            head->prev = NULL;
        else
            tail = NULL;
        delete temp;
    }

    void deleteAtLast() {
        if (head == NULL) {
            cout << "List is empty!!\n";
            return;
        } else if (head == tail) {
            delete head;
            head = tail = NULL;
        } else {
            Snode *temp = tail;
            tail = tail->prev;
            tail->next = NULL;
            delete temp;
        }
    }

    void deleteAtPos(int pos) {
        int count = getCount();
        if (pos < 1 || pos > count) {
            cout << "Invalid position!!\n";
            return;
        }
        if (pos == 1) {
            deleteAtFirst();
        } else if (pos == count) {
            deleteAtLast();
        } else {
            Snode *temp = head;
            for (int i = 1; i < pos; i++) {
                temp = temp->next;
            }
            temp->next->prev = temp->prev;
            temp->prev->next = temp->next;
            delete temp;
        }
    }

    void reverseList() {
        Snode *curr = head, *temp = NULL;
        while (curr != NULL) {
            temp = curr->prev;
            curr->prev = curr->next;
            curr->next = temp;
            curr = curr->prev;
        }
        if (temp != NULL) {
            head = temp->prev;
        }
        cout << "List reversed successfully!!\n";
    }
};

int main() {
    int ch;
    DLinkedList dl;
    while (true) {
        cout << "\n-------- DOUBLY LINKED LIST --------------" << endl;
        cout << "1. Insert data\n2. Display data\n3. Delete data\n4. Search data\n5. Reverse list\n6. Exit\n";
        cin >> ch;

        switch (ch) {
        case 1: dl.insertNode(); cout << "Node Inserted Successfully!!\n"; break;
        case 2: dl.display(); break;
        case 3: dl.deleteNode(); cout << "Node deleted Successfully!!\n"; break;
        case 4: dl.searchNode(); break;
        case 5: dl.reverseList(); break;
        case 6: exit(0);
        default: cout << "\nEnter a proper choice!\n";
        }
    }
    return 0;
}
